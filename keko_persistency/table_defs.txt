CREATE TABLE IF NOT EXISTS loadout (
	LoadoutID INT NOT NULL PRIMARY KEY AUTO_INCREMENT, 
	LoadoutKey CHAR(32) NOT NULL,
	PlayerUID CHAR(100) NOT NULL,
	PlayerName CHAR(255) NOT NULL,
	Time DATETIME NOT NULL,
	Loadout TEXT NOT NULL,
	MedicClass TINYINT NOT NULL,
	EngineerClass TINYINT NOT NULL,
	Rank ENUM('PRIVATE','CORPORAL','SERGEANT','LIEUTENANT','CAPTAIN','MAJOR','COLONEL') NOT NULL,
	Position TEXT NOT NULL,
	UNIQUE (LoadoutKey, PlayerUID))
CHARACTER SET 'utf8'
COLLATE 'utf8_german2_ci';
